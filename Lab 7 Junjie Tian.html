# Calculate the COM and translate to size 28 * 28
for i in range(60):
    imagei = vx.Vx(f'./th20_bi/img{i}_bi.vx')
    im = imagei.i
    com_y, com_x = center_of_mass(im)
    
    s = 28
    tm = np.zeros((s,s), dtype='uint8')
    target_size = 28
    center_y, center_x = target_size // 2, target_size // 2
    
    # Calculate transition
    x_move = 14 - com_x
    y_move = 14 - com_y
    
    # Translate
    for y in range(im.shape[0]):
        for x in range(im.shape[1]):
            new_y = int(y + y_move)
            new_x = int(x + x_move)
            if 0 <= new_y and new_y < s and 0 <= new_x and new_x < s:
                tm[new_y][new_x] = im[y][x]
    
    tmimage = vx.Vx()
    tmimage.i = tm.astype('uint8')
    tmimage.write(f'./idata/img{i}_28.png')

""" Custom datatset loader
    based on https://github.com/utkuozbulak/pytorch-custom-dataset-examples
"""
import pandas as pd
import imageio

class SimpleDataset():
    def __init__(self, data_path, csv_name, transform = None ):
        """
        Args:
            data_path (string): path to the folder where images and csv files are located
            csv_name (string): name of the csv lablel file
            transform: pytorch transforms for transforms and tensor conversion
        """
        # Set path
        self.data_path = data_path
        # Read the csv file
        self.data_info = pd.read_csv(data_path + csv_name, header=None)
        # First column contains the image paths
        self.image_arr = np.asarray(self.data_info.iloc[:, 0])
        # Second column is the labels
        self.label_arr = np.asarray(self.data_info.loc[:, 1])
        # Calculate len
        self.data_len = len(self.data_info.index)
        
    def __getitem__(self, index):
        # Get image name from the pandas df
        single_image_name = self.image_arr[index]
        # Open image
        img_as_img = imageio.imread(self.data_path + single_image_name)
        
        # Get label(class) of the image based on the cropped pandas column
        single_image_label = self.label_arr[index]
        
        return (img_as_img, single_image_label)

    def __len__(self):
        return self.data_len

def KNN_classifier(train_features, train_labels, test_features, test_labels, k):

    start_time = time.time()
    knn.fit(train_features,train_labels)
    train_time = time.time() - start_time
    start_time = time.time()
    print(k)
    print(f'Training time {train_time} seconds for {k}NN')

    score = knn.score(train_features,train_labels)
    test_time = time.time() - start_time
    print(f'Training time {test_time} seconds')
    print(f'Training score with {k}NN is: {score}')
    
    #Evaluate the result
    score = knn.score(test_features,test_labels)
    test_time = time.time() - start_time
    print(f'Test time {test_time} seconds')
    print(f'Test score with {k}NN is: {score}')

1. KNN: If we increase the number of k from 1 to 3, no matter the split ratio is 8:2 or 9:1, the training time is longer while the accuracy is about the same as with k=1. Therefore, it is not worthy if we increase the number of k in this case. 2. PCA: After we apply PCA, we spent some time when processing the steps of PCA. However, the training and testing time are less than before, while the accuracy is about the same. Therefore, it is worthy if we apply PCA before doing KNN classification. 3. MLP: The 2 layer version of MLP takes way longer time than the 1 layer MLP. However, the accuracy is about the same. 4. SVM: The linear version of SVM takes way longer training time than other methods, and the accuracy is not as high as all other models. The reasons may be SVM is sensitive to noise and outliers, rely on linear separability, and sparse data. The cubic polynomial version has a very high accuracy, and the reason may be it allows data to be mapped to a higher-dimensional space, and more complex patterns are captured. The decision boundary is more flexible.
